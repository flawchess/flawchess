```tsx
    {/* Mobile expand: inline MiniBoard */}
    {isExpanded && game.result_fen && (
      <div className="mt-2 flex justify-center sm:hidden" data-testid={`game-card-minimap-${game.game_id}`}>
        <MiniBoard fen={game.result_fen} size={120} flipped={game.user_color === 'black'} />
      </div>
    )}
  </div>
);

// Desktop: wrap in Tooltip if result_fen exists
if (game.result_fen) {
  return (
    <Tooltip>
      <TooltipTrigger asChild>
        {cardContent}
      </TooltipTrigger>
      <TooltipContent side="top" sideOffset={8} className="hidden sm:block p-1 bg-card border border-border">
        <MiniBoard fen={game.result_fen} size={120} flipped={game.user_color === 'black'} />
      </TooltipContent>
    </Tooltip>
  );
}
return cardContent;
```

Key details:
- `TooltipContent` uses `hidden sm:block` to hide on mobile (mobile uses inline expand instead)
- `side="top"` default — Radix auto-flips on viewport collision
- `className="p-1 bg-card border border-border"` overrides default dark tooltip styling
- Card has `cursor-pointer sm:cursor-default` — pointer cursor on mobile only (for tap-expand), default on desktop
- `onClick={() => onToggle?.()}` on the card div handles mobile expand
