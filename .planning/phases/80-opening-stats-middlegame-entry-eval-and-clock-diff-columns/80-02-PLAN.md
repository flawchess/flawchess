       ```python
       @dataclass(slots=True)
       class OpeningMgMetrics:
           """Accumulated middlegame-entry metrics per opening (full_hash key).

           Partition invariant: for every opening with at least one game,
           ``eval_n + mate_n + null_eval_n`` equals the WDL ``total`` returned by
           ``query_position_wdl_batch`` under identical filter params. The three
           buckets are mutually exclusive at the MG-entry row:

           - ``eval_n``      — eval_cp NOT NULL AND eval_mate IS NULL (continuous score)
           - ``mate_n``      — eval_mate IS NOT NULL (forced-mate score; excluded from mean)
           - ``null_eval_n`` — both eval_cp and eval_mate IS NULL (no score available)
           """
           eval_sum: float       # signed user-perspective sum, in centipawns
           eval_sumsq: float     # sum of squared signed eval_cp values
           eval_n: int           # games with eval_cp NOT NULL AND eval_mate IS NULL (continuous-eval rows only)
           mate_n: int           # games with eval_mate IS NOT NULL at MG entry
           null_eval_n: int      # games with both eval_cp and eval_mate IS NULL at MG entry
           clock_diff_sum: float # sum of (user_clock - opp_clock) at entry, in seconds, signed
           base_time_sum: float  # sum of base_time_seconds across the same games — denominator for pct
           clock_diff_n: int     # games with both user_clock and opp_clock NOT NULL at entry
       ```

           time_control: Sequence[str] | None = None,

       SQL shape (write using SQLAlchemy 2.x async `select()` API — never legacy 1.x):
       ```python
       if not hashes:
           return {}

       # Step 1: Per-game MG-entry ply — MIN(ply) WHERE phase == 1, GROUP BY game_id (NOT IN-subquery; planner hang).
       mg_entry_subq = (
           select(
               GamePosition.game_id.label("game_id"),
               func.min(GamePosition.ply).label("entry_ply"),
           )
           .where(
               GamePosition.user_id == user_id,
               GamePosition.phase == 1,
           )
           .group_by(GamePosition.game_id)
           .subquery("mg_entry")
       )

       # Step 2: Dedup (full_hash, game_id) for the opening filter — same as query_position_wdl_batch.
       dedup = (
           select(
               GamePosition.full_hash.label("full_hash"),
               Game.id.label("game_id"),
               Game.user_color.label("user_color"),
               Game.base_time_seconds.label("base_time_seconds"),
           )
           .join(Game, GamePosition.game_id == Game.id)
           .where(
               GamePosition.user_id == user_id,
               GamePosition.full_hash.in_(hashes),
           )
           .distinct(GamePosition.full_hash, Game.id)
       )
       if color is not None:
           dedup = dedup.where(Game.user_color == color)
       dedup = apply_game_filters(
           dedup, time_control, platform, rated, opponent_type, recency_cutoff,
           opponent_gap_min=opponent_gap_min,
           opponent_gap_max=opponent_gap_max,
       )
       dedup_subq = dedup.subquery("dedup")

       # Step 3: JOIN entry-ply row (gp_entry) and entry+1 row (gp_opp) by (game_id, ply).
       # Aliased GamePosition objects.
       gp_entry = aliased(GamePosition, name="gp_entry")
       gp_opp = aliased(GamePosition, name="gp_opp")

       # Sign expression mirrors endgame_service._classify_endgame_bucket convention.
       sign_expr = case((dedup_subq.c.user_color == "white", 1), else_=-1)
       user_eval_expr = sign_expr * gp_entry.eval_cp  # signed centipawns; NULL when eval_cp NULL.
       clock_diff_expr = gp_entry.clock_seconds - gp_opp.clock_seconds

       # Three mutually-exclusive partition predicates over the gp_entry row:
       has_continuous_eval = and_(gp_entry.eval_cp.isnot(None), gp_entry.eval_mate.is_(None))
       has_mate = gp_entry.eval_mate.isnot(None)
       has_null_eval = and_(gp_entry.eval_cp.is_(None), gp_entry.eval_mate.is_(None))

       agg_select = (
           select(
               dedup_subq.c.full_hash,
               # Eval aggregations: only include rows with continuous eval_cp.
               func.coalesce(
                   func.sum(user_eval_expr).filter(has_continuous_eval), 0.0,
               ).label("eval_sum"),
               func.coalesce(
                   func.sum(user_eval_expr * user_eval_expr).filter(has_continuous_eval), 0.0,
               ).label("eval_sumsq"),
               func.count().filter(has_continuous_eval).label("eval_n"),
               func.count().filter(has_mate).label("mate_n"),
               func.count().filter(has_null_eval).label("null_eval_n"),
               # Clock-diff aggregations: only when both clocks present.
               func.coalesce(
                   func.sum(clock_diff_expr).filter(
                       gp_entry.clock_seconds.isnot(None),
                       gp_opp.clock_seconds.isnot(None),
                   ), 0.0,
               ).label("clock_diff_sum"),
               func.coalesce(
                   func.sum(dedup_subq.c.base_time_seconds).filter(
                       gp_entry.clock_seconds.isnot(None),
                       gp_opp.clock_seconds.isnot(None),
                   ), 0.0,
               ).label("base_time_sum"),
               func.count().filter(
                   gp_entry.clock_seconds.isnot(None),
                   gp_opp.clock_seconds.isnot(None),
               ).label("clock_diff_n"),
           )
           .select_from(dedup_subq)
           .join(mg_entry_subq, mg_entry_subq.c.game_id == dedup_subq.c.game_id)
           .join(
               gp_entry,
               (gp_entry.game_id == mg_entry_subq.c.game_id)
               & (gp_entry.ply == mg_entry_subq.c.entry_ply),
           )
           .outerjoin(
               gp_opp,
               (gp_opp.game_id == mg_entry_subq.c.game_id)
               & (gp_opp.ply == mg_entry_subq.c.entry_ply + 1),
           )
           .group_by(dedup_subq.c.full_hash)
       )

       result = await session.execute(agg_select)
       return {
           row[0]: OpeningMgMetrics(
               eval_sum=float(row[1]),
               eval_sumsq=float(row[2]),
               eval_n=int(row[3]),
               mate_n=int(row[4]),
               null_eval_n=int(row[5]),
               clock_diff_sum=float(row[6]),
               base_time_sum=float(row[7]),
               clock_diff_n=int(row[8]),
           )
           for row in result.fetchall()
       }
       ```
