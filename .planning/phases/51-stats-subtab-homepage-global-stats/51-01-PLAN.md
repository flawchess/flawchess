<objective>
Wire the `opponent_type` and `opponent_strength` filter params end-to-end through the backend (`/stats/global` and `/stats/rating-history`) and the frontend API client + hooks layer, so that Plan 04 can enable the filter UI on the Global Stats page without dead filters.

Implements D-18, D-19, D-20 from 51-CONTEXT.md. This is the narrow backend exception to the v1.9 "No backend API changes" rule — two new Query params, two endpoints, no new data models, no migrations, reuses existing `apply_game_filters()`.

Purpose: The Global Stats page today ignores opponent filters, so bot games are mixed into ratings and global WDL. Users expect opponent-strength comparisons ("how do I do vs. stronger players?") and bot exclusion. GSTA-02 needs this wiring to exist before the FilterPanel can enable the filters.

Output: Backend endpoints, service functions, repository functions, API client, and React hooks all accept and pass through the two new filter params consistently. Integration tests prove the router accepts the params without 422. GlobalStats.tsx is minimally updated to read the existing shared filter state's opponentType/opponentStrength so the build stays green and the page begins filtering by the store's current values (which default to `'human'`/`'any'`).

**VISIBLE BEHAVIOR SHIFT:** After this plan, the Global Stats page defaults to excluding bot games (`opponentType='human'` from `DEFAULT_FILTERS`). Previously bot games were included in global WDL and rating history. This matches user expectation but MUST be called out in the PR description. See D-21 and "Claude's Discretion — New filter default values on Global Stats" in 51-CONTEXT.md.

async def query_rating_history(
    session: AsyncSession,
    user_id: int,
    platform: str,
    recency_cutoff: datetime.datetime | None,
    *,
    opponent_type: str = "human",
    opponent_strength: Literal["any", "stronger", "similar", "weaker"] = "any",
) -> list[Row[Any]]:
    """..."""
    user_rating_expr = case(
        (Game.user_color == "white", Game.white_rating),
        else_=Game.black_rating,
    ).label("user_rating")

    date_col = cast(func.timezone("UTC", Game.played_at), Date).label("date")

    stmt = (
        select(date_col, user_rating_expr, Game.time_control_bucket)
        .distinct(date_col, Game.time_control_bucket)
        .where(
            Game.user_id == user_id,
            user_rating_expr.is_not(None),
            Game.played_at.is_not(None),
        )
        .order_by(date_col, Game.time_control_bucket, Game.played_at.desc())
    )

    # Use shared filter helper per CLAUDE.md "Shared Query Filters".
    # platform is wrapped in a single-element list because apply_game_filters

    stmt = apply_game_filters(
        stmt,
        time_control=None,
        platform=[platform],
        rated=None,
        opponent_type=opponent_type,
        recency_cutoff=recency_cutoff,
        opponent_strength=opponent_strength,
    )

import { useQuery } from '@tanstack/react-query';
import { statsApi } from '@/api/client';
import type { Platform, Recency, TimeControl, OpponentType, OpponentStrength } from '@/types/api';

export function useRatingHistory(
  recency: Recency | null,
  platforms: Platform[] | null,
  opponentType: OpponentType,
  opponentStrength: OpponentStrength,
) {
  const normalizedRecency = recency === 'all' ? null : recency;
  // safe: length === 1 check guarantees index 0 exists
  const platform = platforms && platforms.length === 1 ? platforms[0]! : null;
  return useQuery({
    queryKey: ['ratingHistory', normalizedRecency, platform, opponentType, opponentStrength],
    queryFn: () => statsApi.getRatingHistory(normalizedRecency, platform, opponentType, opponentStrength),
  });
}

export function useGlobalStats(
  recency: Recency | null,
  platforms: Platform[] | null,
  opponentType: OpponentType,
  opponentStrength: OpponentStrength,
) {
  const normalizedRecency = recency === 'all' ? null : recency;
  // safe: length === 1 check guarantees index 0 exists
  const platform = platforms && platforms.length === 1 ? platforms[0]! : null;
  return useQuery({
    queryKey: ['globalStats', normalizedRecency, platform, opponentType, opponentStrength],
    queryFn: () => statsApi.getGlobalStats(normalizedRecency, platform, opponentType, opponentStrength),
  });
}

export function useMostPlayedOpenings(filters?: {
  recency: Recency | null;
  timeControls: TimeControl[] | null;
  platforms: Platform[] | null;
  rated: boolean | null;
  opponentType: OpponentType;
  opponentStrength?: string;
}) {
  const normalizedRecency = filters?.recency === 'all' ? null : (filters?.recency ?? null);
  const timeControl = filters?.timeControls ?? null;
  const platform = filters?.platforms ?? null;
  const rated = filters?.rated ?? null;
  const opponentType = filters?.opponentType ?? 'human';
  const opponentStrength = filters?.opponentStrength ?? 'any';

  return useQuery({
    queryKey: ['mostPlayedOpenings', normalizedRecency, timeControl, platform, rated, opponentType, opponentStrength],
    queryFn: () => statsApi.getMostPlayedOpenings({
      recency: normalizedRecency,
      time_control: timeControl,
      platform: platform,
      rated: rated,
      opponent_type: opponentType,
      opponent_strength: opponentStrength,
    }),
  });
}
