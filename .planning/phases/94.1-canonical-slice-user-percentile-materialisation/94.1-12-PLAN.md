async def users_with_zero_pending(
    session: AsyncSession,
    user_ids: Sequence[int],
) -> list[int]:
    """Return the subset of `user_ids` whose pending-eval count is zero.

    Issued as ONE aggregated SQL statement (WR-01 fix). Replaces the per-user
    `count_pending_evals` loop in `eval_drain.py` post-commit Stage B fan-out.
    Construction: an inline VALUES table over `user_ids` LEFT JOIN'd to games
    on (uid = games.user_id AND games.evals_completed_at IS NULL), GROUP BY uid,
    HAVING count(games.id) = 0. This guarantees users with zero rows in `games`
    at all are still returned (LEFT JOIN preserves them; the outer count is 0).

    Args:
        session: AsyncSession (read-only is fine).
        user_ids: Sequence of internal user PKs to check. Empty input short-circuits
            to `[]` without issuing any SQL.

    Returns:
        List of user_ids (subset of input) where the count of games with
        `evals_completed_at IS NULL` is zero. Returned in unspecified order.
        Empty list if no input users have zero pending evals.
    """

#   LEFT JOIN games
#     ON games.user_id = input_uids.uid AND games.evals_completed_at IS NULL
#   GROUP BY input_uids.uid

async with async_session_maker() as read_session:

    )
    affected_user_ids = [row[0] for row in user_id_rows.all()]
    if affected_user_ids:
        zero_pending = await users_with_zero_pending(read_session, affected_user_ids)
        for uid in zero_pending:
            asyncio.create_task(compute_stage_b(uid))
