      params: {
        ...(timeControl ? { time_control: timeControl } : {}),

**3. Create `frontend/src/hooks/useEndgames.ts`**:

```typescript
import { useQuery } from '@tanstack/react-query';
import { endgameApi } from '@/api/client';
import type { TimeControl, Color } from '@/types/api';

export function useEndgameStats(timeControl: TimeControl | null, color: Color | null) {
  return useQuery({
    queryKey: ['endgameStats', timeControl, color],
    queryFn: () => endgameApi.getEndgameStats(timeControl, color),
  });
}

export function useConversionStats(timeControl: TimeControl | null) {
  return useQuery({
    queryKey: ['conversionStats', timeControl],
    queryFn: () => endgameApi.getConversionStats(timeControl),
  });
}
```
