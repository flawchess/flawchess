       ```python
       """Phase 80.1 prod-DB sanity check (D-09). Read-only.

       Usage:
           PROD_TUNNEL_DATABASE_URL=postgresql+asyncpg://...@localhost:15432/... \\
               uv run python scripts/prodcheck_80_1.py 1 42 137

       Requires bin/prod_db_tunnel.sh to be running (forwards prod PG to localhost:15432).
       Prints, per user and per color, the count of OpeningInsightFinding strengths /
       weaknesses surfaced. Compare numbers between a pre-switch checkout (main) and
       the phase 80.1 working tree.
       """
       import asyncio
       import os
       import socket
       import sys

       from sqlalchemy.ext.asyncio import async_sessionmaker, create_async_engine

       from app.services.opening_insights_service import compute_insights


       def _tunnel_open(host: str = "localhost", port: int = 15432) -> bool:
           with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
               s.settimeout(1.0)
               try:
                   s.connect((host, port))
                   return True
               except OSError:
                   return False


       async def _check_user(session_factory, user_id: int) -> None:
           async with session_factory() as session:
               # compute_insights signature mirrors POST /api/insights/openings; pass
               # default filters (no time-control / platform / rated narrowing) so the
               # numbers match a default dashboard load.
               result = await compute_insights(session=session, user_id=user_id)
               sw = len(result.strengths_white)
               ww = len(result.weaknesses_white)
               sb = len(result.strengths_black)
               wb = len(result.weaknesses_black)
               print(
                   f"user={user_id:>6}  "
                   f"white: {sw} strengths / {ww} weaknesses    "
                   f"black: {sb} strengths / {wb} weaknesses    "
                   f"total: {sw + ww + sb + wb}"
               )


       async def main(user_ids: list[int]) -> int:
           if not _tunnel_open():
               print(
                   "ERROR: prod-DB tunnel not open on localhost:15432. "
                   "Run `bin/prod_db_tunnel.sh` first.",
                   file=sys.stderr,
               )
               return 2
           dsn = os.environ.get("PROD_TUNNEL_DATABASE_URL")
           if not dsn:
               print(
                   "ERROR: set PROD_TUNNEL_DATABASE_URL to the prod read-only DSN "
                   "(e.g. postgresql+asyncpg://flawchess_ro:...@localhost:15432/flawchess).",
                   file=sys.stderr,
               )
               return 2
           engine = create_async_engine(dsn, echo=False)
           session_factory = async_sessionmaker(engine, expire_on_commit=False)
           try:
               for uid in user_ids:
                   await _check_user(session_factory, uid)
           finally:
               await engine.dispose()
           return 0


       if __name__ == "__main__":
           if len(sys.argv) < 2:
               print("usage: prodcheck_80_1.py <user_id> [<user_id> ...]", file=sys.stderr)
               sys.exit(2)
           uids = [int(a) for a in sys.argv[1:]]
           sys.exit(asyncio.run(main(uids)))
       ```
