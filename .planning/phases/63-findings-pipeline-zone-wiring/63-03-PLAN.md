    ```python
    """Pydantic v2 schemas for the endgame findings pipeline (Phase 63).

    Defined per FIND-01 / FIND-05 — the schema is the contract consumed by
    insights_service.compute_findings (Plan 04) and by the Phase 65 LLM endpoint's
    prompt-assembly. Field names are LOCKED once this ships: renaming after
    Phase 65 forces a prompt revision.

    Re-exports Zone / Trend / SampleQuality / Window / MetricId / SubsectionId from
    app.services.endgame_zones so consumers have one import path.
    """

    from typing import Literal

    from pydantic import BaseModel, Field

    from app.services.endgame_zones import (
        MetricId,
        SampleQuality,
        SubsectionId,
        Trend,
        Window,
        Zone,
    )

    # Re-export type aliases so callers can `from app.schemas.insights import Zone`
    # without needing two import paths. Do NOT redefine — Plan 01 owns the source.
    __all__ = [
        "EndgameTabFindings",
        "FilterContext",
        "FlagId",
        "MetricId",
        "SampleQuality",
        "SectionId",
        "SubsectionFinding",
        "SubsectionId",
        "Trend",
        "Window",
        "Zone",
    ]

    # Four cross-section flags precomputed deterministically by the findings service
    # (D-09). The Phase 65 prompt references these IDs verbatim, so the set is locked.
    FlagId = Literal[
        "baseline_lift_mutes_score_gap",
        "clock_entry_advantage",
        "no_clock_entry_advantage",
        "notable_endgame_elo_divergence",
    ]

    # Section grouping for Phase 65 prompt assembly (NOT stored on SubsectionFinding
    # — section membership is derived at consumption time per CONTEXT.md §Specifics).
    SectionId = Literal[
        "overall",
        "metrics_elo",
        "time_pressure",
        "type_breakdown",
    ]


    class FilterContext(BaseModel):
        """User's active dashboard filter state, forwarded to compute_findings.

        Mirrors the endgame router's query-parameter surface. Two caveats:

        1. `color` and `rated_only` are carried here for filter-faithfulness but are
           NOT forwarded to endgame_service (which has no color filter). Per INS-03
           they are also NOT fed into the LLM prompt (Phase 65 concern).

        2. `opponent_type` is hardcoded to "human" inside compute_findings; not exposed
           here (Phase 63 does not expose bot-filter findings).

        3. FilterContext.recency is the user's dashboard recency filter — it is SEPARATE
           from the two internal windows (all_time, last_3mo) that compute_findings always
           produces. Per RESEARCH.md §Pitfall 4.
        """

        recency: Literal[
            "all_time", "week", "month", "3months", "6months", "year", "3years", "5years"
        ] = "all_time"
        opponent_strength: Literal["any", "stronger", "similar", "weaker"] = "any"
        color: Literal["all", "white", "black"] = "all"
        time_controls: list[str] = Field(default_factory=list)
        platforms: list[str] = Field(default_factory=list)
        rated_only: bool = False


    class SubsectionFinding(BaseModel):
        """Finding for one subsection × one time window.

        Empty-window convention: when a subsection has zero qualifying games for the
        window, the service emits value=NaN (serialized as JSON null by Pydantic v2),
        zone="typical", trend="n_a", sample_size=0, sample_quality="thin",
        is_headline_eligible=False. Phase 65 prompt-assembly skips findings where
        sample_quality=="thin" AND value is null.

        dimension carries per-combo or per-bucket identity (e.g. {"platform": "chess.com",
        "time_control": "blitz"} for endgame_elo_timeline, {"bucket": "conversion"} for
        endgame_metrics bucketed rows).

        Field declaration order is load-bearing — findings_hash is computed from
        model_dump_json which emits fields in declaration order, re-serialised with
        sort_keys=True via json.dumps for cross-session stability (RESEARCH.md §Hash
        Implementation).
        """

        subsection_id: SubsectionId
        parent_subsection_id: SubsectionId | None = None
        window: Window
        metric: MetricId
        value: float
        zone: Zone
        trend: Trend
        weekly_points_in_window: int
        sample_size: int
        sample_quality: SampleQuality
        is_headline_eligible: bool
        dimension: dict[str, str] | None = None


    class EndgameTabFindings(BaseModel):
        """Deterministic findings for the Endgame tab.

        findings_hash is a 64-char lowercase hex SHA256 of the canonical JSON of
        this model with `as_of` and `findings_hash` itself excluded. Excluding
        `as_of` means identical findings on different days cache-hit (FIND-05).

        findings_hash is populated by compute_findings AFTER all other fields are
        set. On construction the caller passes findings_hash="" as a placeholder;
        the service computes the hash and returns a new model with the hash filled.
        """

        as_of: str  # ISO date, e.g. "2026-04-20"
        filters: FilterContext
        findings: list[SubsectionFinding]
        flags: list[FlagId]
        findings_hash: str
    ```

    **Field order rationale:** `SubsectionFinding` declares `subsection_id` and `parent_subsection_id` first (identity), then window/metric (coordinate), then value/zone/trend (measurement), then sample_size/sample_quality/is_headline_eligible (quality gate), then dimension (fan-out identity, last because it's often None). `EndgameTabFindings` declares as_of first (easy to eyeball), filters next, then payload (findings + flags), then hash last.

    **Why `Field(default_factory=list)` for `time_controls` / `platforms`:** Pydantic v2 disallows mutable defaults in class bodies; using a bare `[]` default triggers a warning/error.

    **`__all__` rationale:** explicit re-export list so `from app.schemas.insights import Zone` works without surprise. Plan 04 imports several names from this module.

    **ty compliance:** Every field annotated; `MetricId`/`SubsectionId`/`Zone`/etc. all come from `endgame_zones` via import (no redeclaration). No `# type: ignore` needed.
