from sqlalchemy.dialects.postgresql import insert as pg_insert
from sqlalchemy.sql import func

async def upsert_percentile(
    session: AsyncSession,
    *,
    user_id: int,
    metric: CdfMetricId,
    value: float,
    percentile: float | None,
    n_games: int,
    cdf_snapshot: date,
) -> None:
    stmt = pg_insert(UserBenchmarkPercentile).values(
        user_id=user_id, metric=metric, value=value,
        percentile=percentile, n_games=n_games, cdf_snapshot=cdf_snapshot,
    )
    stmt = stmt.on_conflict_do_update(
        index_elements=["user_id", "metric"],
        set_={
            "value": stmt.excluded.value,
            "percentile": stmt.excluded.percentile,
            "n_games": stmt.excluded.n_games,
            "cdf_snapshot": stmt.excluded.cdf_snapshot,
            "computed_at": func.now(),
        },
    )
    await session.execute(stmt)

**Own-session background task with Sentry capture** (RESEARCH Pattern 3 / Example 3):

```python
# Source: app/routers/imports.py:67-69 + app/services/eval_drain.py:550-565 + CLAUDE.md Sentry rules
import asyncio
import sentry_sdk
from app.core.database import async_session_maker
from app.services.global_percentile_cdf import interpolate_percentile, CdfMetricId

STAGE_A_METRIC: CdfMetricId = "score_gap"
STAGE_B_METRICS: tuple[CdfMetricId, ...] = (
    "achievable_score_gap", "section2_score_gap_conv", "section2_score_gap_parity",
)

async def compute_stage_a(user_id: int) -> None:
    """Background task — never propagates exceptions back to trigger site."""
    try:
        async with async_session_maker() as session:
            value, n_games = await _compute_score_gap_for_user(session, user_id)
            if value is None:                       # no canonical-slice games → no row
                return
            percentile = interpolate_percentile(STAGE_A_METRIC, value)
            await upsert_percentile(
                session, user_id=user_id, metric=STAGE_A_METRIC,
                value=value, percentile=percentile, n_games=n_games,
                cdf_snapshot=date.today(),
            )
            await session.commit()
    except asyncio.CancelledError:
        raise                                       # lifespan shutdown contract
    except Exception as exc:
        sentry_sdk.set_context("percentile_compute",
                               {"user_id": user_id, "stage": "A"})
        sentry_sdk.capture_exception(exc)
```

`compute_stage_b(user_id)` mirrors the same shape but iterates `STAGE_B_METRICS`.

**DB URL resolution + safety guard** (RESEARCH Example 4, composed from `backfill_eval.py` patterns):

```python
_TARGET_PORT: dict[str, int] = {"dev": 5432, "prod": 15432}
_LOCAL_HOSTS: frozenset[str] = frozenset({"localhost", "127.0.0.1", "::1"})

def _db_url(target: str) -> str:
    override = os.environ.get(f"BACKFILL_{target.upper()}_DB_URL")
    if override:
        host = urlparse(override).hostname
        if host not in _LOCAL_HOSTS:
            raise ValueError(f"override host must be localhost, got {host!r}")
        return override
    port = _TARGET_PORT[target]
    parsed = urlparse(settings.DATABASE_URL)
    new_netloc = f"{parsed.username}:{parsed.password}@localhost:{port}"
    return urlunparse(parsed._replace(netloc=new_netloc))

def _assert_target_safe(url: str, target: str) -> None:
    port_token = f":{_TARGET_PORT[target]}"
    if port_token not in url:
        raise SystemExit(f"Refusing to run: URL does not contain {port_token!r}")
    if target == "prod":
        try:
            socket.create_connection(("localhost", 15432), timeout=2).close()
        except OSError:
            raise SystemExit("Refusing to run: bin/prod_db_tunnel.sh is not up")
```
