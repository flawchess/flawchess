**`_recent_capped_cte` per-TC restriction** (lines 237–259) — the existing shared `recent_capped` CTE partitions `ROW_NUMBER()` by `(g.user_id, g.time_control_bucket)`. For the per-TC builders, inject a `AND g.time_control_bucket = '{tc}'` predicate inside the inner `recent_games` WHERE. **Do NOT modify `_recent_capped_cte` itself** — either (a) write a sibling `_recent_capped_per_tc_cte(snapshot_date, tc)` helper that adds the predicate, or (b) inline the CTE in each new builder. Option (a) is cleaner per the existing module pattern. Quote of the existing helper (lines 237–259):

```python
def _recent_capped_cte(snapshot_date: date | None) -> str:
    recency = _recency_window_sql(snapshot_date)
    return f"""recent_capped AS (
  SELECT g.id, g.user_id, g.user_color, g.result, g.played_at
  FROM (
    SELECT g.*,
           row_number() OVER (PARTITION BY g.user_id, g.time_control_bucket
                              ORDER BY g.played_at DESC) AS rn
    FROM games g
    JOIN selected_users su ON su.user_id = g.user_id
    WHERE g.rated AND NOT g.is_computer_game
      AND g.white_rating IS NOT NULL AND g.black_rating IS NOT NULL
      AND g.played_at >= {recency}
      AND {equal_footing_filter_sql()}
  ) g
  WHERE g.rn <= {RECENT_GAMES_PER_TC_CAP}
)"""
```

```python
def _recent_capped_per_tc_cte(
    snapshot_date: date | None,
    tc: Literal["bullet", "blitz", "rapid", "classical"],
) -> str:
    recency = _recency_window_sql(snapshot_date)
    # PARTITION BY (g.user_id) — only one TC inside the restricted set, so
    # the per-TC partition collapses to per-user. The 1000-cap then means

  SELECT g.id, g.user_id, g.user_color, g.result, g.played_at
  FROM (
    SELECT g.*,

    FROM games g
    JOIN selected_users su ON su.user_id = g.user_id

      AND g.white_rating IS NOT NULL AND g.black_rating IS NOT NULL
      AND g.played_at >= {recency}
      AND {equal_footing_filter_sql()}
      AND g.time_control_bucket = '{tc}'
```

```python
        )
    raise ValueError(f"Unknown metric_id: {metric_id!r}")
```

**Pooled-aggregate `per_user` / `per_user_values` shape** — see the `per_user_cte_section2` example (lines 490–505) for the canonical shape, where `HAVING count(*) >= {SECTION2_MIN_SPANS_PER_BUCKET}` gates inclusion and `per_user_values` emits `(metric_value, n_games)`:

```python
per_user_bucket AS (
  SELECT user_id, bucket,
         avg(gap_span) AS mean_gap,
         count(*) AS span_n
  FROM gap_rows
  WHERE gap_span IS NOT NULL
  GROUP BY user_id, bucket
  HAVING count(*) >= {SECTION2_MIN_SPANS_PER_BUCKET}
),
per_user_values AS (
  SELECT
    mean_gap AS metric_value,
    span_n AS n_games
  FROM per_user_bucket
  WHERE bucket = '{bucket_label}'
)
```

```python
async def compute_stage_b(
    user_id: int,
    *,
    session_maker: async_sessionmaker[AsyncSession] | None = None,
) -> None:
    maker = session_maker if session_maker is not None else _default_session_maker
    try:
        async with maker() as session:
            for metric_id in STAGE_B_METRICS:
                try:
                    result = await _compute_metric_for_user(session, user_id, metric_id)
                    if result is None:
                        # Below pooled floor for this metric → no row.
                        continue
                    value, n_games = result
                    percentile: float | None = interpolate_percentile(metric_id, value)
                    await upsert_percentile(
                        session,
                        user_id=user_id,
                        metric=metric_id,
                        value=value,
                        n_games=n_games,
                        percentile=percentile,
                        cdf_snapshot=date.today(),
                    )
                except asyncio.CancelledError:
                    raise
                except Exception as exc:
                    # Per-metric capture: other metrics continue (plan spec).
                    sentry_sdk.set_context(
                        "percentile_compute",
                        {"user_id": user_id, "stage": "B", "metric": metric_id},
                    )
                    sentry_sdk.capture_exception(exc)
            await session.commit()
    except asyncio.CancelledError:
        raise
    except Exception as exc:
        # Top-level capture for session-open or commit failures.
        sentry_sdk.set_context("percentile_compute", {"user_id": user_id, "stage": "B"})
        sentry_sdk.capture_exception(exc)
```

**`_compute_metric_for_user` signature is unchanged** (lines 88–124) — quote:

```python
async def _compute_metric_for_user(
    session: AsyncSession,
    user_id: int,
    metric_id: CdfMetricId,
) -> tuple[float, int] | None:
    su_cte = selected_users_cte(source="single_user")
    pooled_cte = per_user_cte_for(metric_id, source="single_user", snapshot_date=None)
    sql = (
        f"WITH {su_cte},\n{pooled_cte}\n"
        f"SELECT metric_value::float AS value, n_games::int AS n_games\n"
        f"FROM per_user_values\n"
        f"LIMIT 1"
    )
    result = await session.execute(text(sql).bindparams(user_id=user_id))
    row = result.fetchone()
    if row is None or row.value is None or row.n_games is None:
        return None
    return (float(row.value), int(row.n_games))
```
