    2. Add to `frontend/src/api/client.ts` — add a new `statsApi` named export:
       ```typescript
       import type { RatingHistoryResponse, GlobalStatsResponse } from '@/types/stats';

       export const statsApi = {
         getRatingHistory: (recency: string | null) =>
           apiClient.get<RatingHistoryResponse>('/stats/rating-history', {
             params: recency ? { recency } : {},
           }).then(r => r.data),
         getGlobalStats: (recency: string | null) =>
           apiClient.get<GlobalStatsResponse>('/stats/global', {
             params: recency ? { recency } : {},
           }).then(r => r.data),
       };
       ```

    3. Create `frontend/src/hooks/useStats.ts`:
       ```typescript
       import { useQuery } from '@tanstack/react-query';
       import { statsApi } from '@/api/client';
       import type { Recency } from '@/types/api';

       export function useRatingHistory(recency: Recency | null) {
         const normalizedRecency = recency === 'all' ? null : recency;
         return useQuery({
           queryKey: ['ratingHistory', normalizedRecency],
           queryFn: () => statsApi.getRatingHistory(normalizedRecency),
         });
       }

       export function useGlobalStats(recency: Recency | null) {
         const normalizedRecency = recency === 'all' ? null : recency;
         return useQuery({
           queryKey: ['globalStats', normalizedRecency],
           queryFn: () => statsApi.getGlobalStats(normalizedRecency),
         });
       }
       ```
