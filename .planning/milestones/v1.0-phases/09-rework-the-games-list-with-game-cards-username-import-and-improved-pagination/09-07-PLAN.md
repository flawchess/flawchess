    2. **Add handler** for inline import from the add flow:
       ```tsx
       const handleAdd = async (platform: 'chess.com' | 'lichess') => {
         const trimmed = addUsername.trim();
         if (!trimmed) return;
         try {
           const result = await trigger.mutateAsync({ platform, username: trimmed });
           setAddingPlatform(null);
           setAddUsername('');
           onOpenChange(false);
           onImportStarted(result.job_id);
         } catch {
           toast.error('Failed to start import. Please try again.');
         }
       };
       ```

    3. **Update chess.com platform row** (lines 170-190): After the existing `{profile?.chess_com_username && (...Sync button...)}`, add an else branch:
       ```tsx
       {!profile?.chess_com_username && (
         addingPlatform === 'chess.com' ? (
           <div className="flex items-center gap-2">
             <Input
               type="text"
               placeholder="chess.com username"
               value={addUsername}
               onChange={(e) => setAddUsername(e.target.value)}
               onKeyDown={(e) => e.key === 'Enter' && handleAdd('chess.com')}
               autoFocus
               className="h-8 w-40 text-sm"
               data-testid="import-add-username-chess-com"
             />
             <Button size="sm" onClick={() => handleAdd('chess.com')} disabled={trigger.isPending || !addUsername.trim()} data-testid="btn-add-import-chess-com">

             </Button>
             <Button size="sm" variant="ghost" onClick={() => { setAddingPlatform(null); setAddUsername(''); }} data-testid="btn-add-cancel-chess-com">

             </Button>
           </div>
         ) : (
           <Button size="sm" variant="outline" onClick={() => { setAddingPlatform('chess.com'); setAddUsername(''); }} data-testid="btn-add-chess-com">

         )
       ```
