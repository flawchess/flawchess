Add a small helper function `_normalize_tc_str(tc_str: str) -> str | None` that both normalizers call:
```python
def _normalize_tc_str(tc_str: str) -> str | None:
    """Normalize time control string: drop +0 suffix when increment is 0."""
    if not tc_str or tc_str == "-":
        return None
    if "+" in tc_str:
        base, inc = tc_str.split("+", 1)
        if inc == "0":
            return base
    return tc_str
```

```typescript
function formatTimeControl(tcStr: string): string {
  if (tcStr.includes('+')) {
    const [baseSec, inc] = tcStr.split('+');
    const baseMin = Math.floor(Number(baseSec) / 60);
    return `${baseMin}+${inc}`;
  }
  // No increment — just convert seconds to minutes
  const baseMin = Math.floor(Number(tcStr) / 60);
  return String(baseMin);
}
```
