```python
# AUTO-GENERATED — do not edit by hand.
# Regenerate with: uv run python scripts/gen_endgame_skill_reference.py
# Source: FlawChess Benchmark (Lichess 2300+, ≥10-span floor, equal-footing filter).

REFERENCE_VERSION = "v1"

# Empirical quantile knots of per-user Conv+Parity composite over the FlawChess
# Benchmark population. Shape: list of (composite_value, percentile) sorted ASC.
# Percentile in [0.0, 1.0]; composite_value is a signed float (ΔES scale).
QUANTILE_KNOTS: list[tuple[float, float]] = [
    (-0.20, 0.0),   # p0
    (-0.12, 0.05),  # p5
    # ... ~100 knots ...
    (0.10, 0.95),   # p95
    (0.18, 1.0),    # p100
]


def interpolate_percentile(composite: float) -> float:
    """Map per-user Conv+Parity composite to percentile rank [0.0, 1.0].

    Uses piecewise-linear interpolation between QUANTILE_KNOTS. Clamps to
    [0.0, 1.0] at the tails — equivalent to the _ENDGAME_ELO_SKILL_CLAMP_*
    pattern used by _endgame_elo_from_skill.

    Pure function. No I/O.
    """
    import bisect
    if composite <= QUANTILE_KNOTS[0][0]:
        return 0.0
    if composite >= QUANTILE_KNOTS[-1][0]:
        return 1.0
    vals = [k[0] for k in QUANTILE_KNOTS]
    i = bisect.bisect_right(vals, composite) - 1
    lo_x, lo_p = QUANTILE_KNOTS[i]
    hi_x, hi_p = QUANTILE_KNOTS[i + 1]
    t = (composite - lo_x) / (hi_x - lo_x)
    return lo_p + t * (hi_p - lo_p)
```

```python
    check_mode = "--check" in sys.argv
    content = _render()
    if check_mode:
        if not _OUTPUT.exists():
            print(f"DRIFT: {_OUTPUT.relative_to(_REPO_ROOT)} does not exist.", file=sys.stderr)
            sys.exit(1)
        existing = _OUTPUT.read_text(encoding="utf-8")
        if existing != content:
            print(
                f"DRIFT: {_OUTPUT.relative_to(_REPO_ROOT)} is out of date. "
                "Run `uv run python scripts/gen_endgame_skill_reference.py` to regenerate.",
                file=sys.stderr,
            )
            sys.exit(1)
        print(f"OK: {_OUTPUT.relative_to(_REPO_ROOT)} is up to date.")
    else:
        _OUTPUT.parent.mkdir(parents=True, exist_ok=True)
        _OUTPUT.write_text(content, encoding="utf-8")
        print(f"Wrote {_OUTPUT.relative_to(_REPO_ROOT)}")


if __name__ == "__main__":
    main()
```

```python
def _endgame_skill_percentile_from_window(
    rows: Sequence[Row[Any] | tuple[Any, ...]],
) -> float | None:
    """Conv+Parity ΔES composite → percentile rank for the ELO timeline rolling window.

    Replaces the rate-based _endgame_skill_from_bucket_rows (Phase 87.3).
    Each row is one game's endgame-entry. Gap = game_result_score - ES_entry
    (terminal span — no LEAD column needed; one entry per game in the window).

    Returns s in [0.0, 1.0] via interpolate_percentile, or None when fewer
    than CONFIDENCE_MIN_N spans in every active bucket.
    """
    conv_gaps: list[float] = []
    parity_gaps: list[float] = []

    for row in rows:
        (
            _played_at, _platform, _tc,
            user_color, _white_rating, _black_rating,
            eval_cp, eval_mate, result,
        ) = row
        bucket = _classify_endgame_bucket(eval_cp, eval_mate, user_color)
        if bucket not in ("conversion", "parity"):
            continue
        gap = _compute_span_gap(eval_cp, eval_mate, result, user_color, next_eval_cp=None, next_eval_mate=None)
        if gap is None:
            continue
        if bucket == "conversion":
            conv_gaps.append(gap)
        else:
            parity_gaps.append(gap)

    active_means: list[float] = []
    if len(conv_gaps) >= CONFIDENCE_MIN_N:
        active_means.append(sum(conv_gaps) / len(conv_gaps))
    if len(parity_gaps) >= CONFIDENCE_MIN_N:
        active_means.append(sum(parity_gaps) / len(parity_gaps))

    if not active_means:
        return None
    return interpolate_percentile(sum(active_means) / len(active_means))
```

```python
"""Tests for endgame_skill_reference: interpolate_percentile invariants.

Covers:
- Monotonicity of interpolate_percentile over the QUANTILE_KNOTS range.
- Clamp behavior at tails (below min knot -> 0.0, above max knot -> 1.0).
- Median invariant: the median composite knot maps to approximately 0.5.
- Interior interpolation: a known interior knot value maps to its exact percentile.
"""

import pytest

from app.services.endgame_skill_reference import QUANTILE_KNOTS, interpolate_percentile


class TestInterpolatePercentile:
    """Unit tests for the interpolate_percentile pure function."""

    def test_clamps_below_min_knot(self):
        """Values below the min knot map to 0.0."""
        assert interpolate_percentile(QUANTILE_KNOTS[0][0] - 1.0) == 0.0

    def test_clamps_above_max_knot(self):
        """Values above the max knot map to 1.0."""
        assert interpolate_percentile(QUANTILE_KNOTS[-1][0] + 1.0) == 1.0

    def test_monotonic_over_knot_range(self):
        """interpolate_percentile is non-decreasing over the knot range."""
        n = 200
        lo, hi = QUANTILE_KNOTS[0][0], QUANTILE_KNOTS[-1][0]
        step = (hi - lo) / n
        vals = [interpolate_percentile(lo + i * step) for i in range(n + 1)]
        for i in range(len(vals) - 1):
            assert vals[i] <= vals[i + 1], f"Non-monotonic at index {i}"

    def test_median_invariant(self):
        """The median knot composite maps to approximately 0.5."""
        mid = len(QUANTILE_KNOTS) // 2
        median_composite = QUANTILE_KNOTS[mid][0]
        result = interpolate_percentile(median_composite)
        assert abs(result - 0.5) < 0.05, f"Median invariant violated: {result} != 0.5"

    def test_exact_knot_value(self):
        """A knot's composite_value maps to its exact percentile."""
        test_knot = QUANTILE_KNOTS[len(QUANTILE_KNOTS) // 4]  # p25-ish
        composite, expected_pct = test_knot
        result = interpolate_percentile(composite)
        assert abs(result - expected_pct) < 1e-9, f"Knot interpolation off: {result} != {expected_pct}"
```
