```python
def _compute_clock_diff_timeline(
    clock_rows: Sequence[Row[Any] | tuple[Any, ...]],
    window: int = CLOCK_PRESSURE_TIMELINE_WINDOW,
) -> ClockDiffTimelineResponse:
    """Build the per-ISO-week rolling clock-diff timeline (CONTEXT §2 A-2).

    Pooled across TCs (no time-control faceting). Filters out rows that lack
    clock data, have base_time_seconds <= 0, exceed MAX_CLOCK_PCT_OF_BASE, or
    are missing played_at. One point per ISO Monday using the rolling-mean
    value of the last eligible game in that week.
    """
    # Phase 1: build a flat list of (played_at, clock_diff_pct) for eligible rows.
    eligible: list[tuple[datetime.datetime, float]] = []
    per_week_counts: dict[datetime.date, int] = defaultdict(int)
    for row in clock_rows:
        time_control_bucket = row[1]
        base_time_seconds = row[2]
        user_color = row[5]
        ply_array = row[6]
        clock_array = row[7]
        played_at = row[8]
        if time_control_bucket is None or base_time_seconds is None or base_time_seconds <= 0:
            continue
        if played_at is None:
            continue
        user_clock, opp_clock = _extract_entry_clocks(ply_array, clock_array, user_color)
        if user_clock is None or opp_clock is None:
            continue
        if user_clock > MAX_CLOCK_PCT_OF_BASE * base_time_seconds or opp_clock > MAX_CLOCK_PCT_OF_BASE * base_time_seconds:
            continue
        clock_diff_pct = (user_clock - opp_clock) / base_time_seconds * 100
        eligible.append((played_at, clock_diff_pct))
        monday = played_at.date() - datetime.timedelta(days=played_at.date().weekday())
        per_week_counts[monday] += 1

    # Phase 2: sort by played_at ascending, then rolling-window walk.
    eligible.sort(key=lambda t: t[0])
    rolling: list[float] = []
    week_to_rolling: dict[datetime.date, tuple[float, int]] = {}  # monday -> (rolling_avg, window_size)
    for played_at, diff_pct in eligible:
        rolling.append(diff_pct)
        if len(rolling) > window:
            rolling = rolling[-window:]
        avg = sum(rolling) / len(rolling)
        monday = played_at.date() - datetime.timedelta(days=played_at.date().weekday())
        week_to_rolling[monday] = (round(avg, 4), len(rolling))  # overwrite — last game wins

    # Phase 3: emit in week-ascending order.
    points: list[ClockDiffTimelinePoint] = []
    for monday in sorted(week_to_rolling.keys()):
        avg, n = week_to_rolling[monday]
        points.append(ClockDiffTimelinePoint(
            date=monday.isoformat(),
            avg_clock_diff_pct=avg,
            game_count=n,
            per_week_game_count=per_week_counts[monday],
        ))
    return ClockDiffTimelineResponse(points=points)
```
