GAP-2 — the inactivity break annotation is duplicated inline in two charts and absent from the four other ordinal-axis timelines; a long inactive stretch silently compresses into one ordinal step on those four too. One shared helper makes the annotation honest everywhere and removes the duplicated map.

  export interface InactivityGapReferenceLinesProps {
    dates: string[];            // MUST be ascending-sorted ISO YYYY-MM-DD; caller's responsibility
    yAxisId?: string;           // omit for single-default-axis charts (ScoreChart, RatingChart)
    thresholdDays?: number;     // defaults to INACTIVITY_GAP_THRESHOLD_DAYS via computeInactivityGaps
  }
  // Returns a flat array of <ReferenceLine> elements (NOT a wrapper component).
  // Recharts (ComposedChart/LineChart) discovers ReferenceLine by scanning DIRECT
  // children; a wrapping component/Fragment hides them. The helper is a plain
  // function returning ReactElement[] so the spread `{...inactivityGapReferenceLines(...)}`
  // / `{inactivityGapReferenceLines(...)}` inside the chart yields direct children.
  export function inactivityGapReferenceLines(props: InactivityGapReferenceLinesProps): ReactElement[];
