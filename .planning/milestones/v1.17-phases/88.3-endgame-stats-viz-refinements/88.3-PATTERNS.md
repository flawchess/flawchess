**Core function signature pattern** (model from `signedBandGradient.ts`):
```typescript
// signedBandGradient exports one function with typed params and explicit return type.
// inactivityGaps.ts copies the same structure:
export function computeInactivityGaps(
  sortedDates: string[],
  thresholdDays: number = INACTIVITY_GAP_THRESHOLD_DAYS,
): InactivityGap[] {
  const gaps: InactivityGap[] = [];
  for (let i = 0; i < sortedDates.length - 1; i++) {
    const a = new Date(sortedDates[i]!);     // noUncheckedIndexedAccess: ! is safe (loop bound)
    const b = new Date(sortedDates[i + 1]!);
    const days = Math.round((b.getTime() - a.getTime()) / 86400000);
    if (days > thresholdDays) {
      const label = days >= 365
        ? `ã${(days / 365).toFixed(1)}y inactive`
        : `ã${Math.round(days / 30)}mo inactive`;
      gaps.push({ afterIndex: i, gapDays: days, label });
    }
  }
  return gaps;
}
```

**`noUncheckedIndexedAccess` compliance:** The `!` non-null assertion on `sortedDates[i]!` and `sortedDates[i + 1]!` is provably in-bounds because the loop runs while `i < sortedDates.length - 1`. Add a brief comment to document this is intentional, following the codebase convention seen in `EndgameEloTimelineSection.tsx`.
